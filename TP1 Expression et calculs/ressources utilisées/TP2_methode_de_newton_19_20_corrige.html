def mon_newton1(f, x0, fp, e=1e-8, e2=1e-6, kmax=50):
    """Retourne une valeur x telle que f(x)=0 avec une prcision e sur x.
    fp(x) retourne la drive de f en x.
    La valeur initiale est x0.
    kmax est le nombre d'itrations maximum.
    S'il est atteint ou si une drive quasi nulle est trouve (abs(fp(x)) < e2),
    la fonction retourne 'ERREUR kmax' ou 'ERREUR drive'.
    Si tout va bien, mon_newton1 affiche la valeur de x recherche."""
    # A COMPLETER
    
    x = x0
    k = 0
    dx = f(x) / fp(x)     # dx = x(k) - x(k+1)
    
    while abs(dx) > e and k < kmax :
        if abs(fp(x)) < e2 :
            return print('ERREUR drive')   # return sort de la fonction 'mon_newton'
        else :
            x = x - dx
            dx = f(x) / fp(x)
            k = k + 1
        if k == kmax :
            return print('ERREUR kmax')
    print("Nombre d'itrations :", k)
    return x

Principe d'utilisation de la mthode de Newton :

La mthode de Newton ncessite d'avoir une ide de la solution pour dfinir la valeur initiale, indispensable s'il existe plusieurs solutions. De ce fait, son utilisation usuelle consiste :

  1. tracer graphiquement la fonction,
  2. utiliser la mthode de Newton partir d'un point initial dfini par l'tude graphique,
  3. ajouter la ou les solutions sur le graphique pour vrification.

Cette dmarche sera effectuer de faon quasi-systmatique dans les exemples suivants.

def mon_newton2(f, x0, h=1e-6, e=1e-8, e2=1e-6, kmax=50):
    """Retourne une valeur x telle que f(x)=0 avec une prcision e sur x.
    La drive est estime par diffrence finie centre avec un pas h.
    La valeur initiale est x0.
    kmax est le nombre d'itrations maximum.
    S'il est atteint ou si une drive quasi nulle est trouve (abs(fp(x)) < e2),
    la fonction retourne 'ERREUR kmax' ou 'ERREUR drive'.
    Si tout va bien, mon_newton affiche la valeur de x recherche."""
    # A COMPLETER

    
    x = x0
    k = 0
    fp = (f(x+h)-f(x-h)) / (2*h)
    dx = f(x) / fp                        # dx = x(k) - x(k+1)
    
    while abs(dx) > e and k < kmax :
        if abs(fp) < e2 :
            return print('ERREUR drive')   # return sort de la fonction 'mon_newton'
        else :
            x = x - dx
            fp = (f(x+h)-f(x-h))/(2*h)
            dx = f(x) / fp
            k = k + 1
        if k == kmax :
            return print('ERREUR kmax')
    print("Nombre d'itrations :", k)
    return x

def fermeture(alpha,beta):
    """Retourne la valeur de l'quation de fermeture.
    alpha et beta doivent tre en radians.
    La valeur est nulle si l'quation est vrifie."""
    r = e * (a * np.cos(beta) - b * np.sin(beta)) + d * e * np.cos(alpha-beta) \
      - e * c * np.sin(alpha - beta) + (a * c + d * b) * np.sin(alpha) \
      + (b * c - a * d) * np.cos(alpha) - (a ** 2 + b ** 2 + c ** 2 + d ** 2)/2
    return r
